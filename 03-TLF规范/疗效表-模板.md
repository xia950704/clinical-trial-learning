# 疗效表-模板 (Efficacy Tables Template)

## 概述

## 2. R 代码模板

```r
# =============================================================================
# 疗效分析脚本 (Efficacy Analysis)
# 依赖包: {survival}, {survminer}, {gtsummary}, {tidyverse}
# 作者: [姓名] | 日期: [YYYY-MM-DD] | 版本: v1.0
# =============================================================================

library(survival)
library(survminer)
library(gtsummary)
library(tidyverse)
library(dplyr)
library(tidyr)
library(fmsb)  # 用于 Clopper-Pearson 置信区间

# ---- 0. 环境设置 ----
options(survprint.method = "table")
options(gtsummary.default_pkg_style = list(
  style_number = list(digits = 1, big.mark = ","),
  style_percent = list(digits = 1)
))

# ---- 1. 数据加载 ----

# 1.1 肿瘤反应数据
tumor_response <- read_csv("data/tumor_response.csv",
                           col_types = cols(
                             patient_id = col_character(),
                             treatment = col_factor(levels = c("Control", "Experimental")),
                             best_response = col_factor(levels = c("CR", "PR", "SD", "PD", "NE")),
                             response_date = col_date(),
                             baseline_tumor = col_double(),
                             nadir_tumor = col_double()
                           ))

# 1.2 生存数据
survival_data <- read_csv("data/survival_data.csv",
                          col_types = cols(
                            patient_id = col_character(),
                            treatment = col_factor(levels = c("Control", "Experimental")),
                            pfs_time = col_double(),
                            pfs_event = col_integer(),
                            os_time = col_double(),
                            os_event = col_integer(),
                            pfs_start = col_date(),
                            pfs_end = col_date(),
                            os_start = col_date(),
                            os_end = col_date(),
                            censor_reason = col_character()
                          ))

# 合并数据
clinical_data <- survival_data %>%
  left_join(tumor_response %>% select(patient_id, best_response, baseline_tumor, nadir_tumor),
            by = "patient_id")

# ---- 2. 客观缓解率 (ORR) 分析 ----

# 2.1 响应汇总
response_summary <- clinical_data %>%
  filter(!is.na(best_response)) %>%  # 排除无法评估的患者
  group_by(treatment, best_response) %>%
  summarise(n = n(), .groups = "drop") %>%
  group_by(treatment) %>%
  mutate(
    total_evaluable = sum(n),
    pct = n / total_evaluable * 100
  ) %>%
  ungroup()

print(response_summary)

# 2.2 计算 ORR 及 95% CI (Clopper-Pearson 方法)
orr_calculation <- clinical_data %>%
  filter(!is.na(best_response)) %>%
  group_by(treatment) %>%
  summarise(
    n_evaluable = n(),
    n_cr = sum(best_response == "CR", na.rm = TRUE),
    n_pr = sum(best_response == "PR", na.rm = TRUE),
    n_orr = n_cr + n_pr,
    orr_pct = n_orr / n_evaluable * 100,
    # Clopper-Pearson 置信区间
    ci_lower = bincron(n_orr, n_evaluable)[1],
    ci_upper = bincron(n_orr, n_evaluable)[3],
    .groups = "drop"
  )

print(orr_calculation)

# 2.3 ORR 森林图数据
orr_forest_data <- orr_calculation %>%
  mutate(
    orr_display = sprintf("%.1f%% (%.1f%%-%.1f%%)", orr_pct, ci_lower, ci_upper),
    log_orr = log(orr_pct / 100),
    se_log_orr = sqrt((1 - orr_pct/100) / (n_evaluable * orr_pct/100))
  )

# ---- 3. 无进展生存期 (PFS) 分析 ----

# 3.1 Kaplan-Meier 估计
pfs_km <- survfit(
  Surv(pfs_time, pfs_event) ~ treatment,
  data = clinical_data,
  conf.type = "log-log"
)

# 3.2 输出 KM 表
pfs_km_table <- summary(pfs_km, times = c(3, 6, 9, 12, 18, 24)) %>%
  tidy() %>%
  mutate(
    time = times,
    n_risk = n,
    n_event = n.event,
    survival = surv,
    lower_ci = conf.lower,
    upper_ci = conf.upper
  ) %>%
  select(time, n_risk, n_event, survival, lower_ci, upper_ci)

print(pfs_km_table)

# 3.3 中位 PFS 及 95% CI
pfs_median <- survfit(Surv(pfs_time, pfs_event) ~ treatment, data = clinical_data) %>%
  broom::tidy() %>%
  filter(estimate == "median") %>%
  select(treatment, estimate, lower, upper) %>%
  rename(
    treatment = treatment,
    median_pfs = estimate,
    ci_lower = lower,
    ci_upper = upper
  )

print(pfs_median)

# 3.4 Log-rank 检验
pfs_logrank <- survdiff(
  Surv(pfs_time, pfs_event) ~ treatment,
  data = clinical_data
)

pfs_pvalue <- 1 - pchisq(pfs_logrank$chisq, length(pfs_logrank$n) - 1)
pfs_hr <- summary(coxph(Surv(pfs_time, pfs_event) ~ treatment, data = clinical_data))$coefficients

pfs_results <- tibble(
  treatment = c("Control", "Experimental"),
  median_pfs = pfs_median$median_pfs,
  ci_lower = pfs_median$ci_lower,
  ci_upper = pfs_median$ci_upper,
  n_events = c(sum(clinical_data$pfs_event[clinical_data$treatment == "Control"]),
               sum(clinical_data$pfs_event[clinical_data$treatment == "Experimental"])),
  hr = pfs_hr["treatmentExperimental", "exp(coef)"],
  hr_ci_lower = pfs_hr["treatmentExperimental", "lower .95"],
  hr_ci_upper = pfs_hr["treatmentExperimental", "upper .95"],
  p_value = pfs_pvalue
)

print(pfs_results)

# ---- 4. 总生存期 (OS) 分析 ----

# 4.1 Kaplan-Meier 估计
os_km <- survfit(
  Surv(os_time, os_event) ~ treatment,
  data = clinical_data,
  conf.type = "log-log"
)

# 4.2 输出 KM 表
os_km_table <- summary(os_km, times = c(6, 12, 18, 24, 36, 48)) %>%
  tidy() %>%
  mutate(
    time = times,
    n_risk = n,
    n_event = n.event,
    survival = surv,
    lower_ci = conf.lower,
    upper_ci = conf.upper
  ) %>%
  select(time, n_risk, n_event, survival, lower_ci, upper_ci)

print(os_km_table)

# 4.3 中位 OS 及 95% CI
os_median <- survfit(Surv(os_time, os_event) ~ treatment, data = clinical_data) %>%
  broom::tidy() %>%
  filter(estimate == "median") %>%
  select(treatment, estimate, lower, upper) %>%
  rename(
    treatment = treatment,
    median_os = estimate,
    ci_lower = lower,
    ci_upper = upper
  )

print(os_median)

# 4.4 Log-rank 检验
os_logrank <- survdiff(
  Surv(os_time, os_event) ~ treatment,
  data = clinical_data
)

os_pvalue <- 1 - pchisq(os_logrank$chisq, length(os_logrank$n) - 1)
os_hr <- summary(coxph(Surv(os_time, os_event) ~ treatment, data = clinical_data))$coefficients

os_results <- tibble(
  treatment = c("Control", "Experimental"),
  median_os = os_median$median_os,
  ci_lower = os_median$ci_lower,
  ci_upper = os_median$ci_upper,
  n_events = c(sum(clinical_data$os_event[clinical_data$treatment == "Control"]),
               sum(clinical_data$os_event[clinical_data$treatment == "Experimental"])),
  hr = os_hr["treatmentExperimental", "exp(coef)"],
  hr_ci_lower = os_hr["treatmentExperimental", "lower .95"],
  hr_ci_upper = os_hr["treatmentExperimental", "upper .95"],
  p_value = os_pvalue
)

print(os_results)

# ---- 5. Cox 比例风险模型 ----

# 5.1 单因素 Cox 模型
cox_pfs_univariate <- coxph(Surv(pfs_time, pfs_event) ~ treatment, data = clinical_data)
cox_os_univariate <- coxph(Surv(os_time, os_event) ~ treatment, data = clinical_data)

# 5.2 多因素 Cox 模型 (根据研究设计调整协变量)
# 示例协变量：年龄、性别、ECOG 评分、疾病分期
cox_pfs_multivariate <- coxph(
  Surv(pfs_time, pfs_event) ~ treatment + age + sex + ecog + stage,
  data = clinical_data,
  na.action = na.exclude
)

cox_os_multivariate <- coxph(
  Surv(os_time, os_event) ~ treatment + age + sex + ecog + stage,
  data = clinical_data,
  na.action = na.exclude
)

# 5.3 模型摘要
summary(cox_pfs_multivariate)
summary(cox_os_multivariate)

# 5.4 比例风险假设检验
cox_zph_pfs <- cox.zph(cox_pfs_multivariate)
cox_zph_os <- cox.zph(cox_os_multivariate)

print(cox_zph_pfs)
print(cox_zph_os)

# ---- 6. 生存曲线图 ----

# 6.1 PFS Kaplan-Meier 曲线
pfs_km_plot <- ggsurvplot(
  pfs_km,
  data = clinical_data,
  risk.table = TRUE,
  risk.table.col = "strata",
  risk.table.height = 0.25,
  pval = TRUE,
  pval.method = TRUE,
  conf.int = TRUE,
  conf.int.style = "ribbon",
  palette = c("#2E86AB", "#A23B72"),
  legend.title = "Treatment",
  legend.labs = c("Control", "Experimental"),
  xlab = "Time (months)",
  ylab = "Progression-Free Survival Probability",
  ggtheme = theme_bw(),
  font.main = c(14, "bold"),
  font.x = c(12, "plain"),
  font.y = c(12, "plain"),
  font.tickslab = c(10, "plain"),
  surv.median.line = "hv",
  break.time.by = 3,
  tables.theme = theme_cleantable()
)

ggsave("output/PFS_KM_curve.png", pfs_km_plot$plot, width = 8, height = 6, dpi = 300)
ggsave("output/PFS_KM_curve_full.png", pfs_km_plot, width = 10, height = 8, dpi = 300)

# 6.2 OS Kaplan-Meier 曲线
os_km_plot <- ggsurvplot(
  os_km,
  data = clinical_data,
  risk.table = TRUE,
  risk.table.col = "strata",
  risk.table.height = 0.25,
  pval = TRUE,
  pval.method = TRUE,
  conf.int = TRUE,
  conf.int.style = "ribbon",
  palette = c("#2E86AB", "#A23B72"),
  legend.title = "Treatment",
  legend.labs = c("Control", "Experimental"),
  xlab = "Time (months)",
  ylab = "Overall Survival Probability",
  ggtheme = theme_bw(),
  font.main = c(14, "bold"),
  font.x = c(12, "plain"),
  font.y = c(12, "plain"),
  font.tickslab = c(10, "plain"),
  surv.median.line = "hv",
  break.time.by = 6,
  tables.theme = theme_cleantable()
)

ggsave("output/OS_KM_curve.png", os_km_plot$plot, width = 8, height = 6, dpi = 300)
ggsave("output/OS_KM_curve_full.png", os_km_plot, width = 10, height = 8, dpi = 300)

# ---- 7. 森林图 (Forest Plot) ----

# 7.1 ORR 森林图
orr_forest <- orr_forest_data %>%
  ggplot(aes(x = orr_pct, y = treatment)) +
  geom_point(size = 4, color = "#2E86AB") +
  geom_errorbarh(aes(xmin = ci_lower, xmax = ci_upper), height = 0.3, color = "#2E86AB") +
  geom_vline(xintercept = 0, linetype = "dashed", color = "gray") +
  labs(
    title = "Objective Response Rate (ORR)",
    subtitle = "With 95% Confidence Intervals",
    x = "ORR (%)",
    y = "Treatment Group"
  ) +
  theme_minimal() +
  theme(
    plot.title = element_text(size = 14, face = "bold"),
    plot.subtitle = element_text(size = 11),
    axis.text = element_text(size = 11),
    axis.title = element_text(size = 12)
  )

ggsave("output/ORR_forest_plot.png", orr_forest, width = 8, height = 4, dpi = 300)

# 7.2 HR 森林图
hr_forest_data <- bind_rows(
  pfs_results %>% select(treatment, hr, hr_ci_lower, hr_ci_upper) %>% mutate(endpoint = "PFS"),
  os_results %>% select(treatment, hr, hr_ci_lower, hr_ci_upper) %>% mutate(endpoint = "OS")
) %>%
  mutate(
    hr_label = sprintf("%.2f (%.2f-%.2f)", hr, hr_ci_lower, hr_ci_upper)
  )

hr_forest <- hr_forest_data %>%
  ggplot(aes(x = hr, y = paste(endpoint, treatment, sep = " - "))) +
  geom_point(size = 4, color = "#A23B72") +
  geom_errorbarh(aes(xmin = hr_ci_lower, xmax = hr_ci_upper), height = 0.3, color = "#A23B72") +
  geom_vline(xintercept = 1, linetype = "dashed", color = "gray", linewidth = 1) +
  labs(
    title = "Hazard Ratios for PFS and OS",
    subtitle = "Experimental vs Control",
    x = "Hazard Ratio (95% CI)",
    y = ""
  ) +
  scale_x_log10() +
  theme_minimal() +
  theme(
    plot.title = element_text(size = 14, face = "bold"),
    plot.subtitle = element_text(size = 11),
    axis.text = element_text(size = 11),
    axis.title = element_text(size = 12),
    panel.grid.minor = element_blank()
  )

ggsave("output/HR_forest_plot.png", hr_forest, width = 8, height = 5, dpi = 300)

# ---- 8. 输出汇总表格 ----

# 8.1 ORR 汇总
orr_table <- orr_calculation %>%
  mutate(
    cr_pct = n_cr / n_evaluable * 100,
    pr_pct = n_pr / n_evaluable * 100,
    orr_ci = sprintf("[%.1f%%, %.1f%%]", ci_lower, ci_upper)
  ) %>%
  select(treatment, n_evaluable, n_cr, cr_pct, n_pr, pr_pct, n_orr, orr_pct, orr_ci)

write_csv(orr_table, "output/ORR_summary.csv")

# 8.2 PFS 汇总
write_csv(pfs_results, "output/PFS_summary.csv")

# 8.3 OS 汇总
write_csv(os_results, "output/OS_summary.csv")

# 8.4 Cox 模型结果
cox_pfs_summary <- broom::tidy(cox_pfs_multivariate, conf.int = TRUE, exponentiate = TRUE)
cox_os_summary <- broom::tidy(cox_os_multivariate, conf.int = TRUE, exponentiate = TRUE)

write_csv(cox_pfs_summary, "output/Cox_PFS_multivariate.csv")
write_csv(cox_os_summary, "output/Cox_OS_multivariate.csv")

# ---- 9. 运行日志 ----
run_log <- list(

  output_files = c(

  ),

saveRDS(run_log, "output/run_log_efficacy.rds")

message("✅ 疗效分析完成！")
```

## 4. 注意事项

### 4.1 分析人群定义

| 人群 | 定义 | 用途 |
|:---|:---|:---|
| ITT (意向治疗) | 所有随机化患者 | 主要疗效分析 |
| PP (符合方案) | 完成研究且无重大方案违背的患者 | 敏感性分析 |
| Safety | 接受至少一剂研究药物 | 安全性分析 |

### 4.2 删失处理

- **PFS**: 从随机化到疾病进展或死亡（以先发生者为准），无事件则删失于末次肿瘤评估日
- **OS**: 从随机化到任何原因死亡，存活患者删失于末次随访日

### 4.3 统计方法选择

| 场景 | 推荐方法 |
|:---|:---|
| 主要终点比较 | Log-rank 检验 + Cox 模型 |
| 分层随机化 | 分层 Log-rank 检验 + 分层 Cox 模型 |
| 非比例风险 | 加权 Log-rank 检验 (如 Fleming-Harrington) |
| 亚组分析 | 交互作用检验 (treatment × subgroup) |

### 4.4 亚组分析

```r
# 亚组森林图数据准备
subgroup_analysis <- clinical_data %>%
  group_by(treatment, age_group, sex, ecog, stage) %>%
  summarise(
    n = n(),
    events = sum(pfs_event),
    median_pfs = quantile(pfs_time[pfs_event == 1], 0.5, na.rm = TRUE),
    .groups = "drop"
  )

# 交互作用检验
cox_interaction <- coxph(
  Surv(pfs_time, pfs_event) ~ treatment * age_group + treatment * sex,
  data = clinical_data,
  na.action = na.exclude
)
summary(cox_interaction)
```
