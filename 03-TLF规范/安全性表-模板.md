## 2. R 代码模板

```r
# =============================================================================
# 安全性分析脚本 (Safety Analysis)
# 依赖包: {gtsummary}, {tidyverse}, {dplyr}
# 作者: [姓名] | 日期: [YYYY-MM-DD] | 版本: v1.0
# =============================================================================

library(gtsummary)
library(tidyverse)
library(dplyr)
library(tidyr)

# ---- 0. 环境设置 ----
options(gtsummary.default_pkg_style = list(
  style_number = list(digits = 0, big.mark = ","),
  style_percent = list(digits = 1)
))

# ---- 1. 数据加载 ----
# 不良事件数据
ae_data <- read_csv("data/adverse_events.csv",
                    col_types = cols(
                      patient_id = col_character(),
                      treatment = col_factor(levels = c("Control", "Experimental")),
                      ae_term = col_character(),
                      ae_soc = col_character(),
                      ae_grade = col_integer(),
                      ae_related = col_factor(levels = c("Not Related", "Possibly Related", "Probably Related", "Definitely Related")),
                      ae_serious = col_factor(levels = c("No", "Yes")),
                      ae_sae = col_factor(levels = c("No", "Yes")),
                      ae_outcome = col_factor(levels = c("Recovered", "Recovering", "Not Recovered", "Recovered with Sequelae", "Fatal", "Unknown"))
                    ))

# 患者基线数据
baseline_data <- read_csv("data/clinical_baseline.csv")

# ---- 2. 患者水平汇总 ----

# 2.1 任何级别 AE
ae_patient_summary <- ae_data %>%
  group_by(patient_id, treatment) %>%
  summarise(
    any_ae = ifelse(n() > 0, 1, 0),
    grade_3_plus = ifelse(any(ae_grade >= 3), 1, 0),
    treatment_related = ifelse(any(ae_related %in% c("Possibly Related", "Probably Related", "Definitely Related")), 1, 0),
    serious_ae = ifelse(any(ae_serious == "Yes"), 1, 0),
    sae = ifelse(any(ae_sae == "Yes"), 1, 0),
    discontinued = ifelse(any(ae_outcome == "Fatal" | ae_related == "Definitely Related"), 1, 0),
    fatal = ifelse(any(ae_outcome == "Fatal"), 1, 0),
    .groups = "drop"
  ) %>%
  group_by(treatment) %>%
  summarise(
    n_patients = n(),
    any_ae_n = sum(any_ae),
    any_ae_pct = mean(any_ae) * 100,
    grade_3_plus_n = sum(grade_3_plus),
    grade_3_plus_pct = mean(grade_3_plus) * 100,
    treatment_related_n = sum(treatment_related),
    treatment_related_pct = mean(treatment_related) * 100,
    serious_ae_n = sum(serious_ae),
    serious_ae_pct = mean(serious_ae) * 100,
    sae_n = sum(sae),
    sae_pct = mean(sae) * 100,
    discontinued_n = sum(discontinued),
    discontinued_pct = mean(discontinued) * 100,
    fatal_n = sum(fatal),
    fatal_pct = mean(fatal) * 100
  )

print(ae_patient_summary)

# 2.2 格式化为 gtsummary 表格
ae_summary_table <- ae_patient_summary %>%
  pivot_longer(
    cols = -treatment,
    names_to = c(".value", "stat"),
    names_pattern = "(.*)_(n|pct)"
  ) %>%
  mutate(stat = ifelse(stat == "n", "n", "pct")) %>%
  pivot_wider(
    names_from = stat,
    values_from = c(n, pct)
  ) %>%
  mutate(label = case_when(
    n == "any_ae_n" ~ "任何级别 AE",
    n == "grade_3_plus_n" ~ "≥ 3 级 AE",
    n == "treatment_related_n" ~ "治疗相关 AE",
    n == "serious_ae_n" ~ "严重 AE (SAE)",
    n == "sae_n" ~ "严重不良事件 (SAE)",
    n == "discontinued_n" ~ "导致停药 AE",
    n == "fatal_n" ~ "导致死亡 AE"
  )) %>%
  select(treatment, label, n, pct) %>%
  distinct() %>%
  arrange(match(label, c("任何级别 AE", "≥ 3 级 AE", "治疗相关 AE", "严重 AE (SAE)", "导致停药 AE", "导致死亡 AE")))

# ---- 3. 按 SOC 汇总 AE ----

ae_by_soc <- ae_data %>%
  group_by(treatment, ae_soc) %>%
  summarise(
    n_events = n(),
    n_patients = n_distinct(patient_id),
    .groups = "drop"
  ) %>%
  group_by(treatment) %>%
  mutate(pct_patients = n_patients / n_distinct(ae_data$patient_id) * 100) %>%
  ungroup() %>%
  pivot_wider(
    names_from = treatment,
    values_from = c(n_patients, pct_patients),
    names_sep = "_"
  )

print(ae_by_soc)

# ---- 4. 按 MedDRA 术语汇总 AE ----

ae_by_term <- ae_data %>%
  group_by(treatment, ae_term, ae_grade) %>%
  summarise(n_events = n(), .groups = "drop") %>%
  group_by(treatment, ae_term) %>%
  summarise(
    total_events = sum(n_events),
    max_grade = max(ae_grade),
    n_patients = n_distinct(ae_data$patient_id[ae_data$ae_term == ae_term & ae_data$treatment == treatment]),
    .groups = "drop"
  ) %>%
  mutate(
    pct = n_patients / n_distinct(ae_data$patient_id) * 100
  ) %>%
  arrange(desc(n_patients))

# 输出前 20 个最常见 AE
ae_top20 <- ae_by_term %>%
  group_by(treatment) %>%
  top_n(20, n_patients) %>%
  ungroup()

print(ae_top20)

# ---- 5. 实验室异常汇总 ----

lab_data <- read_csv("data/lab_results.csv",
                     col_types = cols(
                       patient_id = col_character(),
                       treatment = col_factor(levels = c("Control", "Experimental")),
                       lab_test = col_character(),
                       visit = col_factor(levels = c("Baseline", "Cycle 1 Day 1", "Cycle 1 Day 15", "Cycle 2 Day 1", "End of Treatment")),
                       value = col_double(),
                       normal_low = col_double(),
                       normal_high = col_double(),
                       grade = col_integer()
                     ))

# 5.1 实验室异常分级汇总 (CTCAE 标准)
lab_abnormality <- lab_data %>%
  filter(!is.na(value)) %>%
  mutate(
    abnormal = ifelse(value < normal_low | value > normal_high, 1, 0),
    grade_3_plus = ifelse(grade >= 3, 1, 0)
  ) %>%
  group_by(treatment, lab_test) %>%
  summarise(
    n_patients = n_distinct(patient_id),
    any_abnormal_n = sum(abnormal),
    any_abnormal_pct = mean(abnormal) * 100,
    grade_3_plus_n = sum(grade_3_plus),
    grade_3_plus_pct = mean(grade_3_plus) * 100,
    mean_value = mean(value, na.rm = TRUE),
    sd_value = sd(value, na.rm = TRUE),
    .groups = "drop"
  )

print(lab_abnormality)

# 5.2 实验室指标变化（从基线到结束）
lab_change <- lab_data %>%
  pivot_wider(
    names_from = visit,
    values_from = value,
    values_fn = mean
  ) %>%
  mutate(
    change_from_baseline = `End of Treatment` - `Baseline`,
    pct_change = (change_from_baseline / `Baseline`) * 100
  )

print(lab_change)

# ---- 6. 生命体征汇总 ----

vital_data <- read_csv("data/vital_signs.csv",
                       col_types = cols(
                         patient_id = col_character(),
                         treatment = col_factor(levels = c("Control", "Experimental")),
                         visit = col_factor(levels = c("Screening", "Baseline", "Cycle 1 Day 1", "Cycle 1 Day 15", "Cycle 2 Day 1", "End of Treatment")),
                         sbp = col_double(),  # 收缩压
                         dbp = col_double(),  # 舒张压
                         hr = col_double(),   # 心率
                         temp = col_double(), # 体温
                         weight = col_double() # 体重
                       ))

# 6.1 生命体征描述性统计
vital_summary <- vital_data %>%
  group_by(treatment, visit) %>%
  summarise(
    n = n(),
    sbp_mean = mean(sbp, na.rm = TRUE),
    sbp_sd = sd(sbp, na.rm = TRUE),
    dbp_mean = mean(dbp, na.rm = TRUE),
    dbp_sd = sd(dbp, na.rm = TRUE),
    hr_mean = mean(hr, na.rm = TRUE),
    hr_sd = sd(hr, na.rm = TRUE),
    temp_mean = mean(temp, na.rm = TRUE),
    temp_sd = sd(temp, na.rm = TRUE),
    weight_mean = mean(weight, na.rm = TRUE),
    weight_sd = sd(weight, na.rm = TRUE),
    .groups = "drop"
  )

print(vital_summary)

# 6.2 生命体征从基线变化
vital_change <- vital_data %>%
  group_by(treatment, visit) %>%
  summarise(
    sbp_mean = mean(sbp, na.rm = TRUE),
    dbp_mean = mean(dbp, na.rm = TRUE),
    hr_mean = mean(hr, na.rm = TRUE),
    weight_mean = mean(weight, na.rm = TRUE),
    .groups = "drop"
  ) %>%
  pivot_wider(
    names_from = visit,
    values_from = c(sbp_mean, dbp_mean, hr_mean, weight_mean)
  ) %>%
  mutate(
    sbp_change = sbp_mean_End.of.Treatment - sbp_mean_Baseline,
    dbp_change = dbp_mean_End.of.Treatment - dbp_mean_Baseline,
    hr_change = hr_mean_End.of.Treatment - hr_mean_Baseline,
    weight_change = weight_mean_End.of.Treatment - weight_mean_Baseline
  )

print(vital_change)

# ---- 7. 输出表格 ----

# 7.1 患者水平 AE 汇总表格
ae_patient_table <- ae_patient_summary %>%
  select(treatment, any_ae_n, any_ae_pct, grade_3_plus_n, grade_3_plus_pct,
         treatment_related_n, treatment_related_pct, serious_ae_n, serious_ae_pct,
         sae_n, sae_pct, discontinued_n, discontinued_pct, fatal_n, fatal_pct) %>%
  mutate(
    any_ae = paste0(any_ae_n, " (", round(any_ae_pct, 1), "%)"),
    grade_3_plus = paste0(grade_3_plus_n, " (", round(grade_3_plus_pct, 1), "%)"),
    treatment_related = paste0(treatment_related_n, " (", round(treatment_related_pct, 1), "%)"),
    serious_ae = paste0(serious_ae_n, " (", round(serious_ae_pct, 1), "%)"),
    sae = paste0(sae_n, " (", round(sae_pct, 1), "%)"),
    discontinued = paste0(discontinued_n, " (", round(discontinued_pct, 1), "%)"),
    fatal = paste0(fatal_n, " (", round(fatal_pct, 1), "%)")
  ) %>%
  select(treatment, any_ae, grade_3_plus, treatment_related, serious_ae, sae, discontinued, fatal)

# 7.2 导出
write_csv(ae_patient_summary, "output/ae_patient_summary.csv")
write_csv(ae_by_soc, "output/ae_by_soc.csv")
write_csv(ae_top20, "output/ae_top20.csv")
write_csv(lab_abnormality, "output/lab_abnormality.csv")
write_csv(vital_summary, "output/vital_summary.csv")

# ---- 8. 运行日志 ----
run_log <- list(
  script_name = "safety_analysis.R",
  execution_date = Sys.Date(),
  input_files = c("data/adverse_events.csv", "data/lab_results.csv", "data/vital_signs.csv"),
  output_files = c(
    "output/ae_patient_summary.csv",
    "output/ae_by_soc.csv",
    "output/ae_top20.csv",
    "output/lab_abnormality.csv",
    "output/vital_summary.csv"
  ),
  packages_used = c("gtsummary", "tidyverse", "dplyr", "tidyr"),
  notes = "Safety analysis tables generated successfully."
)

saveRDS(run_log, "output/run_log_safety.rds")

message("✅ 安全性分析表生成完成！")
```

## 4. 注意事项

1. **MedDRA 编码**: AE 术语应使用 MedDRA 标准字典编码
2. **CTCAE 分级**: 实验室异常应按 CTCAE (Common Terminology Criteria for Adverse Events) 版本分级
3. **治疗相关性**: 由研究者评估，通常分为：无关、可能无关、可能相关、肯定相关
4. **严重性 vs 严重程度**: 严重性 (seriousness) 基于事件结果（如住院、危及生命），严重程度 (severity) 基于分级（1-5 级）
5. **安全人群定义**: 通常包括所有接受至少一剂研究药物的患者（Safety Population）
